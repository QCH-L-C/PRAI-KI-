from flask import Blueprint, jsonify, request
transaction_blueprint = Blueprint('transaction', __name__)
@transaction_blueprint.route('/transactions/new', methods=['POST'])
def new_transaction():
    values = request.get_json()
    required = ['sender', 'recipient', 'amount']
    if not all(k in values for k in required):
        return 'Missing values', 400
    response = {'message': 'Transaction being added'}
    return jsonify(response), 201

4. Utilities

Helper modules for efficiency and security:

Blockchain Utils (blockchain_utils.py)

import hashlib
import json

def hash_block(block):
    """Generates the SHA256 hash of a block."""
    block_string = json.dumps(block, sort_keys=True).encode()
    return hashlib.sha256(block_string).hexdigest()

Encryption (encryption.py)

from cryptography.fernet import Fernet

def generate_key():
    """Generates a new Fernet key."""
    return Fernet.generate_key()

def encrypt_message(message, key):
    """Encrypts a message with the given key."""
    fernet = Fernet(key)
    encrypted = fernet.encrypt(message.encode())
    return encrypted

def decrypt_message(encrypted_message, key):
    """Decrypts a message with the given key."""
    fernet = Fernet(key)
    decrypted = fernet.decrypt(encrypted_message).decode()
    return decrypted

Validation (validation.py)

def validate_transaction_data(data):
    """Validates if the required keys are present in the transaction data."""
    required_keys = ['sender', 'recipient', 'amount']
    return all(key in data for key in required_keys)
